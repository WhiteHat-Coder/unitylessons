1. MonoBehaviour Lifecycle 🔄

Concept:

MonoBehaviour = base class for all Unity scripts

Unity calls specific methods at different times in the game loop

🕐 Event Execution Order:

Method When Called Use For
Awake() When script first loads Initialize references, setup
Start() Before first frame Initialization after Awake
Update() Every frame Input, movement, game logic
FixedUpdate() Fixed time intervals Physics calculations
LateUpdate() After all Updates Camera following, finalization
Example: Lifecycle Methods in Action
using UnityEngine;

public class LifecycleExample : MonoBehaviour
{
    void Awake()
    {
        Debug.Log("1. Awake - Called first");
    }

    void Start()
    {
        Debug.Log("2. Start - Called before first Update");
    }

    void Update()
    {
        Debug.Log("3. Update - Called every frame (~60 times/second)");
    }

    void FixedUpdate()
    {
        Debug.Log("4. FixedUpdate - Called at fixed intervals (50 times/second)");
    }

    void LateUpdate()
    {
        Debug.Log("5. LateUpdate - Called after all Updates");
    }
}
💡 Best Practices:
• Awake() for finding references between scripts
• Start() for initialization that depends on other objects
• Update() for input and regular updates
• FixedUpdate() for physics (moving with Rigidbody)
• LateUpdate() for camera following

2. Unity Input System ⌨️🖱️🎮

Concept:

Input = reading keyboard, mouse, and gamepad input

Use Input class methods in Update()

📝 Keyboard Input:

Example: Basic Keyboard Controls
using UnityEngine;

public class PlayerController : MonoBehaviour
{
    void Update()
    {
        // Check if key is held down
        if (Input.GetKey(KeyCode.W))
        {
            Debug.Log("Moving forward");
        }

        // Check if key was pressed this frame
        if (Input.GetKeyDown(KeyCode.Space))
        {
            Debug.Log("Jump!");
        }

        // Check if key was released this frame
        if (Input.GetKeyUp(KeyCode.LeftShift))
        {
            Debug.Log("Stopped sprinting");
        }
    }
}

🎹 Input Methods:

Method When True Example Use
GetKey() While key is held Continuous movement
GetKeyDown() Frame key is pressed Jump, shoot, interact
GetKeyUp() Frame key is released Stop charging attack

📝 Axis Input (Smooth Movement):

Example: WASD/Arrow Key Movement
public class PlayerMovement : MonoBehaviour
{
    public float speed = 5f;

    void Update()
    {
        // Get input (-1 to 1)
        float horizontal = Input.GetAxis("Horizontal"); // A/D or Left/Right
        float vertical = Input.GetAxis("Vertical"); // W/S or Up/Down

        // Move the player
        Vector3 movement = new Vector3(horizontal, 0, vertical);
        transform.position += movement * speed * Time.deltaTime;

        Debug.Log("H: " + horizontal + ", V: " + vertical);
    }
}

📝 Mouse Input:

Example: Mouse Clicks and Position
void Update()
{
    // Left mouse button (0), Right (1), Middle (2)
    if (Input.GetMouseButtonDown(0))
    {
        Debug.Log("Left click!");
    }

    // Get mouse position
    Vector3 mousePos = Input.mousePosition;
    Debug.Log("Mouse at: " + mousePos);

    // Mouse wheel
    float scroll = Input.GetAxis("Mouse ScrollWheel");
    if (scroll != 0)
    {
        Debug.Log("Scrolling: " + scroll);
    }
}
⚠️ Important: Multiply movement by Time.deltaTime to make it frame-rate independent!
Without it, movement speed varies on different computers.

3. GameObject & Transform 📍

Concept:

GameObject = every object in your scene

Transform = position, rotation, and scale (every GameObject has one)

📝 Manipulating Transform:

Example: Moving, Rotating, Scaling
public class TransformExample : MonoBehaviour
{
    void Update()
    {
        // POSITION
        transform.position = new Vector3(0, 0, 0); // Set exact position
        transform.position += Vector3.up * Time.deltaTime; // Move up
        transform.Translate(Vector3.forward * 5f * Time.deltaTime); // Move forward

        // ROTATION
        transform.rotation = Quaternion.identity; // Reset rotation
        transform.Rotate(0, 90 * Time.deltaTime, 0); // Rotate Y-axis
        transform.eulerAngles = new Vector3(0, 45, 0); // Set exact angles

        // SCALE
        transform.localScale = new Vector3(2, 2, 2); // Double size
        transform.localScale = Vector3.one * 0.5f; // Half size
    }
}

📝 Finding GameObjects:

Example: Find and Reference Objects
public class FindExample : MonoBehaviour
{
    GameObject player;
    Transform target;

    void Start()
    {
        // Find by name
        player = GameObject.Find("Player");

        // Find by tag
        GameObject enemy = GameObject.FindWithTag("Enemy");

        // Find all with tag
        GameObject[] allEnemies = GameObject.FindGameObjectsWithTag("Enemy");
        Debug.Log("Found " + allEnemies.Length + " enemies");

        // Get component from GameObject
        Rigidbody rb = player.GetComponent<Rigidbody>();
    }
}

📝 Practical Example - Follow Player:

Example: Camera Follow Script
public class CameraFollow : MonoBehaviour
{
    public Transform player;
    public Vector3 offset = new Vector3(0, 5, -10);

    void LateUpdate() // LateUpdate for camera!
    {
        if (player != null)
        {
            // Follow player with offset
            transform.position = player.position + offset;

            // Look at player
            transform.LookAt(player);
        }
    }
}

4. Instantiate & Destroy 🏗️💥

Concept:

Instantiate = create new GameObjects at runtime

Destroy = remove GameObjects from the scene

📝 Instantiate Objects:

Example: Spawn Enemies
public class EnemySpawner : MonoBehaviour
{
    public GameObject enemyPrefab; // Assign in Inspector
    public Transform spawnPoint;

    void Start()
    {
        // Spawn at specific position
        Vector3 spawnPos = new Vector3(0, 0, 0);
        Instantiate(enemyPrefab, spawnPos, Quaternion.identity);

        // Spawn at transform location
        Instantiate(enemyPrefab, spawnPoint.position, spawnPoint.rotation);

        // Spawn and keep reference
        GameObject newEnemy = Instantiate(enemyPrefab, transform.position, Quaternion.identity);
        newEnemy.name = "Spawned Enemy";
    }
}

📝 Destroy Objects:

Example: Remove Objects
public class DestroyExample : MonoBehaviour
{
    void Update()
    {
        // Destroy immediately
        if (Input.GetKeyDown(KeyCode.X))
        {
            Destroy(gameObject); // Destroy self
        }

        // Destroy after delay
        Destroy(gameObject, 3f); // Destroy after 3 seconds
    }
}

// Destroy another object
public class Bullet : MonoBehaviour
{
    void OnCollisionEnter(Collision collision)
    {
        // Destroy the object we hit
        Destroy(collision.gameObject);

        // Destroy self
        Destroy(gameObject);
    }
}

📝 Complete Example - Shooting System:

Example: Spawn Bullets on Click
public class PlayerShooting : MonoBehaviour
{
    public GameObject bulletPrefab;
    public Transform firePoint;
    public float bulletSpeed = 20f;

    void Update()
    {
        if (Input.GetButtonDown("Fire1")) // Left mouse
        {
            Shoot();
        }
    }

    void Shoot()
    {
        // Create bullet
        GameObject bullet = Instantiate(bulletPrefab, firePoint.position, firePoint.rotation);

        // Add velocity
        Rigidbody rb = bullet.GetComponent<Rigidbody>();
        rb.velocity = firePoint.forward * bulletSpeed;

        // Auto-destroy after 2 seconds
        Destroy(bullet, 2f);
    }
}

5. Collisions and Triggers 💥

Concept:

Collision = physics-based collisions (objects bounce off each other)

Trigger = detection zones (objects pass through)

Both require Collider components!

Collision vs Trigger:

Collision Trigger
Objects physically collide Objects pass through
Use for walls, obstacles Use for pickups, zones
Requires Rigidbody Check "Is Trigger" on Collider

📝 Collision Events:

Example: Collision Detection
public class CollisionExample : MonoBehaviour
{
    // Called when collision starts
    void OnCollisionEnter(Collision collision)
    {
        Debug.Log("Hit: " + collision.gameObject.name);

        // Check by tag
        if (collision.gameObject.tag == "Enemy")
        {
            Debug.Log("Hit enemy!");
        }
    }

    // Called every frame during collision
    void OnCollisionStay(Collision collision)
    {
        Debug.Log("Still touching " + collision.gameObject.name);
    }

    // Called when collision ends
    void OnCollisionExit(Collision collision)
    {
        Debug.Log("Stopped touching " + collision.gameObject.name);
    }
}

📝 Trigger Events:

Example: Pickup System
public class CoinPickup : MonoBehaviour

            // Add to player score (example)

📝 Complete Example - Damage System:

Example: Take Damage on Hit
public class PlayerHealth : MonoBehaviour
{
    public int maxHealth = 100;
    private int currentHealth;

    void Start()
    {
        currentHealth = maxHealth;
    }

    void OnCollisionEnter(Collision collision)
    {
        if (collision.gameObject.tag == "Enemy")
        {

        }
    }

    void Die()
    {

    }
}

6. Coroutines ⏱️

Concept:

Coroutine = function that can pause execution and resume later

Perfect for timers, delays, and animations

📝 Basic Coroutine:

Example: Simple Delay
using System.Collections;

public class CoroutineExample : MonoBehaviour
{
    void Start()
    {
        Debug.Log("Starting coroutine...");
        StartCoroutine(WaitAndPrint());
        Debug.Log("Coroutine started (but not finished yet)");
    }

    IEnumerator WaitAndPrint()
    {
        Debug.Log("Waiting 2 seconds...");
        
        yield return new WaitForSeconds(2f);
        
        Debug.Log("2 seconds passed!");
    }
}
Output:
Starting coroutine...
Waiting 2 seconds...
Coroutine started (but not finished yet)
(2 seconds later...)
2 seconds passed!

📝 Countdown Timer:

Example: Countdown from 3
IEnumerator Countdown()
{
    for (int i = 3; i > 0; i--)
    {
        Debug.Log(i);
        yield return new WaitForSeconds(1f);
    }
    Debug.Log("GO!");
}

📝 Practical Example - Power-Up Timer:

Example: Temporary Speed Boost
public class PowerUp : MonoBehaviour
{
    public float normalSpeed = 5f;
    public float boostSpeed = 10f;
    public float boostDuration = 5f;
    private float currentSpeed;

    void Start()
    {
        currentSpeed = normalSpeed;
    }

    void OnTriggerEnter(Collider other)
    {
        if (other.CompareTag("SpeedBoost"))
        {
            StartCoroutine(ActivateSpeedBoost());
            Destroy(other.gameObject);
        }
    }

    IEnumerator ActivateSpeedBoost()
    {
        Debug.Log("Speed boost activated!");
        currentSpeed = boostSpeed;

        yield return new WaitForSeconds(boostDuration);

        currentSpeed = normalSpeed;
        Debug.Log("Speed boost ended");
    }
}

💡 Common Yield Types:

  • yield return null; → Wait one frame
  • yield return new WaitForSeconds(time); → Wait X seconds
  • yield return new WaitUntil(() => condition); → Wait until condition true
  • yield return new WaitWhile(() => condition); → Wait while condition true
